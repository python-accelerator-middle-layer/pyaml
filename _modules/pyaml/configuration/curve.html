    @classmethod
    def inverse(cls, curve:np.array) -> np.array:
        """
        Returns the inverse curve.
        Basically swap x and y and sort y in ascending order.

        Parameters
        ----------
        curve : np.array
            Curve to be inverted
        """
        __curve = curve
        __sortedCurve = __curve[__curve[:,1].argsort()]
        return __sortedCurve[:,[1,0]]
